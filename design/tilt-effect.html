Move your cursor across any card — the tilt follows the pointer and pivots from the top edge, so lifted cards never slide under a header or nav bar.

1 · HTML

<!-- add class="tilt" to any card -->
<div class="card tilt">…</div>

2 · CSS

.tilt {
  transition: transform .3s cubic-bezier(.2,.7,.3,1),
              box-shadow .3s ease, border-color .3s ease;
  transform-style: preserve-3d;
  will-change: transform;
}
.tilt:hover {
  box-shadow: 0 18px 50px -22px rgba(139,92,246,.5);
  border-color: rgba(139,92,246,.32);
}

3 · JavaScript

// tune these two for strength
const MAX_X = 10, MAX_Y = 13, SCALE = 1.045;

function bindTilt(el) {
  el.addEventListener('mousemove', (e) => {
    const r = el.getBoundingClientRect();
    const px = (e.clientX - r.left) / r.width;
    const py = (e.clientY - r.top) / r.height;
    const rx = (0.5 - py) * MAX_X;   // tilt up/down
    const ry = (px - 0.5) * MAX_Y;   // tilt left/right
    el.style.transformOrigin = '50% 0%';  // pivot from TOP → no clip
    el.style.transform =
      `perspective(780px) rotateX(${rx}deg) rotateY(${ry}deg) scale(${SCALE})`;
    el.style.zIndex = '20';              // float above siblings
  });
  el.addEventListener('mouseleave', () => {
    el.style.transform = '';
    el.style.zIndex = '';
    el.style.transformOrigin = '';
  });
}

document.querySelectorAll('.tilt').forEach(bindTilt);
Why it never gets cut off: transformOrigin: '50% 0%' pins the pivot to each card's top edge, so it tilts and scales downward instead of rising up. Combined with no translateY, the top never moves — a card in the first row can't slide under a fixed header or nav bar.
Tuning: raise MAX_X/MAX_Y (try 14/18) for a stronger tilt, or lower them (6/8) for subtle. Drop SCALE to 1 if you don't want the card to grow.
